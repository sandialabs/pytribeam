106def laser_state_to_db(state: tbt.LaserState) -> dict:
107    """
108    This function converts a `LaserState` object into a flattened dictionary representation.
109
110    Parameters
111    ----------
112    state : tbt.LaserState
113        The laser state object to convert.
114
115    Returns
116    -------
117    dict
118        A flattened dictionary representation of the laser state.
119    """
120    db = {}
121
122    db["wavelength_nm"] = state.wavelength_nm
123    db["frequency_khz"] = state.frequency_khz
124    db["pulse_divider"] = state.pulse_divider
125    db["pulse_energy_uj"] = state.pulse_energy_uj
126    db["objective_position_mm"] = state.objective_position_mm
127    db["expected_pattern_duration_s"] = state.expected_pattern_duration_s
128
129    beam_shift = state.beam_shift_um
130    db["beam_shift_um_x"] = beam_shift.x
131    db["beam_shift_um_y"] = beam_shift.y
132
133    # we can name these differently depending on the needs of the GUI
134    pattern = state.pattern
135    db["laser_pattern_mode"] = pattern.mode.value
136    db["laser_pattern_rotation_deg"] = pattern.rotation_deg
137    db["laser_pattern_pulses_per_pixel"] = pattern.pulses_per_pixel
138    db["laser_pattern_pixel_dwell_ms"] = pattern.pixel_dwell_ms
139
140    geometry = pattern.geometry
141    db["passes"] = geometry.passes
142    db["laser_scan_type"] = geometry.scan_type.value
143    db["geometry_type"] = geometry.type.value
144
145    if geometry.type == tbt.LaserPatternType.BOX:
146        db["size_x_um"] = geometry.size_x_um
147        db["size_y_um"] = geometry.size_y_um
148        db["pitch_x_um"] = geometry.pitch_x_um
149        db["pitch_y_um"] = geometry.pitch_y_um
150        db["coordinate_ref"] = geometry.coordinate_ref
151
152    if geometry.type == tbt.LaserPatternType.LINE:
153        db["size_um"] = geometry.size_um
154        db["pitch_um"] = geometry.pitch_um
155
156    return db

164

166    -------

169    """
170    connect_msg = "Connection test successful.\n"
171    laser_status = io.StringIO()
172    try:
173        with contextlib.redirect_stdout(laser_status):
174            tfs_laser.TestConnection()

183def _device_connections() -> tbt.DeviceStatus:
184    """
185    Check the connection status of the laser and associated external devices.
186
187    This function checks the connection status of the laser, EBSD, and EDS devices. It is meant to be a quick tool for the GUI and does not provide additional information for troubleshooting.
188
189    Returns
190    -------
191    tbt.DeviceStatus
192        The connection status of the laser, EBSD, and EDS devices.
193    """
194    # laser must be connected to connect with other devices:
195    if not laser_connected():
196        laser = tbt.RetractableDeviceState.ERROR
197        ebsd = tbt.RetractableDeviceState.ERROR
198        eds = tbt.RetractableDeviceState.ERROR
199    else:
200        laser = tbt.RetractableDeviceState.CONNECTED
201        ebsd = devices.connect_EBSD()  # retractable device state
202        eds = devices.connect_EDS()  # retractable device state
203
204    return tbt.DeviceStatus(
205        laser=laser,
206        ebsd=ebsd,
207        eds=eds,
208    )

239def pulse_energy_uj(
240    energy_uj: float,
241    energy_tol_uj: float = Constants.laser_energy_tol_uj,
242    delay_s: float = 3.0,
243) -> bool:
244    """
245    Set the pulse energy on the laser.
246
247    This function sets the pulse energy on the laser and verifies that it has been set correctly. It should be done after setting the pulse divider.
248
249    Parameters
250    ----------
251    energy_uj : float
252        The pulse energy to set in microjoules.
253    energy_tol_uj : float, optional
254        The tolerance for the pulse energy in microjoules (default is Constants.laser_energy_tol_uj).
255    delay_s : float, optional
256        The delay in seconds after setting the pulse energy (default is 3.0 seconds).
257
258    Returns
259    -------
260    bool
261        True if the pulse energy is set correctly.
262
263    Raises
264    ------
265    ValueError
266        If the pulse energy cannot be set correctly.
267    """
268    tfs_laser.Laser_SetPulseEnergy_MicroJoules(energy_uj)
269    time.sleep(delay_s)
270    laser_state = factory.active_laser_state()
271    if not ut.in_interval(
272        val=laser_state.pulse_energy_uj,
273        limit=tbt.Limit(
274            min=energy_uj - energy_tol_uj,
275            max=energy_uj + energy_tol_uj,
276        ),
277        type=tbt.IntervalType.CLOSED,
278    ):
279        raise ValueError(
280            f"Could not properly set pulse energy, requested '{energy_uj}' uJ",
281            f"Current settings is {round(laser_state.pulse_energy_uj,3)} uJ",
282        )
283    return True

323def set_wavelength(
324    wavelength: tbt.LaserWavelength,
325    frequency_khz: float = 60,  # make constnat
326    timeout_s: int = 20,  # 120,  # make constant
327    num_attempts: int = 2,  # TODO make a constant
328    delay_s: int = 5,  # make a constant
329) -> bool:
330    """
331    Set the wavelength and frequency of the laser.
332
333    This function sets the wavelength and frequency of the laser and verifies that they have been set correctly.
334
335    Parameters
336    ----------
337    wavelength : tbt.LaserWavelength
338        The wavelength to set.
339    frequency_khz : float, optional
340        The frequency to set in kHz (default is 60 kHz).
341    timeout_s : int, optional
342        The timeout in seconds for each attempt (default is 20 seconds).
343    num_attempts : int, optional
344        The number of attempts to set the wavelength and frequency (default is 2).
345    delay_s : int, optional
346        The delay in seconds between checks (default is 5 seconds).
347
348    Returns
349    -------
350    bool
351        True if the wavelength and frequency are set correctly, False otherwise.
352    """
353
354    def correct_preset(laser_state: tbt.LaserState):
355        if laser_state.wavelength_nm == wavelength:
356            return math.isclose(laser_state.frequency_khz, frequency_khz, rel_tol=0.05)
357            # TODO use constant for tolerance):
358        return False
359
360    for _ in range(num_attempts):
361        if correct_preset(factory.active_laser_state()):
362            return True
363        print("Adjusting preset...")
364        tfs_laser.Laser_SetPreset(
365            wavelength_nm=wavelength.value, frequency_kHz=frequency_khz
366        )
367        time_remaining = timeout_s
368        while time_remaining > 0:
369            laser_state = factory.active_laser_state()
370            # print(time_remaining, laser_state.frequency_khz)
371            if correct_preset(laser_state=laser_state):
372                return True
373            time.sleep(delay_s)
374            time_remaining -= delay_s
375
376    return False

380    """
381    Measure the laser power in watts.
382
383    This function measures the laser power using an external power meter.

471def pulse_polarization(
472    polarization: tbt.LaserPolarization, wavelength: tbt.LaserWavelength
473) -> bool:
474    """
475    Configure the polarization of the laser light.
476
477    This function sets the polarization of the laser light based on the specified polarization and wavelength. The polarization is controlled via "FlipperConfiguration", which takes the following values:
478        - Waveplate_None switches to Vert. (P)
479        - Waveplate_1030 switches to Horiz. (S)
480        - Waveplate_515 switches to Horiz. (S)
481
482    Parameters
483    ----------
484    polarization : tbt.LaserPolarization
485        The desired polarization of the laser light.
486    wavelength : tbt.LaserWavelength
487        The wavelength of the laser light.
488
489    Returns
490    -------
491    bool
492        True if the polarization is set correctly.
493
494    Raises
495    ------
496    KeyError
497        If the laser wavelength or pulse polarization is invalid.
498    """
499    if polarization == tbt.LaserPolarization.VERTICAL:
500        tfs_laser.FlipperConfiguration("Waveplate_None")
501        return True
502    elif polarization == tbt.LaserPolarization.HORIZONTAL:
503        match_db = {
504            tbt.LaserWavelength.NM_1030: "Waveplate_1030",
505            tbt.LaserWavelength.NM_515: "Waveplate_515",
506        }
507        try:
508            tfs_laser.FlipperConfiguration(match_db[wavelength])
509        except KeyError:
510            raise KeyError(
511                f"Invalid laser wavelength, valid options are {[i.value for i in tbt.LaserWavelength]}"
512            )
513        return True
514    else:
515        raise KeyError(
516            f"Invalid pulse polarization, valid options are {[i.value for i in tbt.LaserPolarization]}"
517        )

567

569
570    Parameters
571    ----------
572    position_mm : float

590
591    if not ut.in_interval(
592        val=position_mm,
593        limit=Constants.laser_objective_limit_mm,
594        type=tbt.IntervalType.CLOSED,
595    ):

598        )
599
600    for _ in range(2):
601        if ut.in_interval(
602            val=tfs_laser.LIP_GetZPosition(),
603            limit=tbt.Limit(
604                min=position_mm - tolerance_mm, max=position_mm + tolerance_mm
605            ),
606            type=tbt.IntervalType.CLOSED,
607        ):
608            return True
609        tfs_laser.LIP_SetZPosition(position_mm, asynchronously=False)
610
611    raise SystemError(
612        f"Unable to move laser injection port objective to requested position of {position_mm} +/- {tolerance_mm} mm.",
613        f"Currently at {tfs_laser.LIP_GetZPosition()} mm.",
614    )

620    tolerance: float,
621    axis: str,
622) -> bool:
623    """
624    Helper function for beam shift.
625
626    This function adjusts the beam shift for the specified axis to the target value within the given tolerance.
627
628    Parameters
629    ----------
630    target : float
631        The target value for the beam shift.
632    current : float
633        The current value of the beam shift.
634    tolerance : float
635        The tolerance for the beam shift.
636    axis : str
637        The axis to adjust ("X" or "Y").
638
639    Returns
640    -------
641    bool
642        True if the beam shift is adjusted to the target value correctly, False otherwise.
643    """
644    for _ in range(2):
645        if ut.in_interval(
646            val=current,
647            limit=tbt.Limit(
648                min=target - tolerance,
649                max=target + tolerance,
650            ),
651            type=tbt.IntervalType.CLOSED,
652        ):
653            return True
654        if axis == "X":
655            tfs_laser.BeamShift_Set_X(value=target)
656            current = tfs_laser.BeamShift_Get_X()
657        if axis == "Y":
658            tfs_laser.BeamShift_Set_Y(value=target)
659            current = tfs_laser.BeamShift_Get_Y()
660
661    return False
