 70def application_files(microscope: tbt.Microscope) -> List[str]:
 71    """
 72    Get the list of application files from the current microscope.
 73
 74    This function retrieves the list of application files available on the current microscope, removes any "None" entries, and sorts the list.
 75
 76    Parameters
 77    ----------
 78    microscope : tbt.Microscope
 79        The microscope object from which to retrieve the application files.
 80
 81    Returns
 82    -------
 83    List[str]
 84        A sorted list of application files available on the microscope.
 85    """
 86    apps = microscope.patterning.list_all_application_files()
 87
 88    # Remove "None" entry
 89    while "None" in apps:
 90        apps.remove("None")
 91    apps.sort(key=str.casefold)
 92
 93    return apps

 96def shutter_control(microscope: tbt.Microscope) -> None:
 97    """
 98    Ensure auto control is set on the e-beam shutter. Manual control is not currently offered.
 99
100    This function checks if the e-beam protective shutter is installed and sets its mode to automatic if it is not already set. If the shutter cannot be set to automatic mode, a SystemError is raised.
101
102    Parameters
103    ----------
104    microscope : tbt.Microscope
105        The microscope object for which to control the e-beam shutter.
106
107    Raises
108    ------
109    SystemError
110        If the e-beam shutter is installed but cannot be set to automatic mode.
111
112    Warnings
113    --------
114    UserWarning
115        If the e-beam shutter is not installed or if it is set to automatic mode.
116    """
117    shutter = microscope.beams.electron_beam.protective_shutter
118    if not shutter.is_installed:
119        warnings.warn("Protective E-beam shutter not installed on this system.")
120        return
121    status = shutter.mode.value
122    if status != tbt.ProtectiveShutterMode.AUTOMATIC:
123        shutter.mode.value = tbt.ProtectiveShutterMode.AUTOMATIC
124    new_status = shutter.mode.value
125    if new_status != tbt.ProtectiveShutterMode.AUTOMATIC:
126        raise SystemError(
127            "E-beam shutter for FIB milling is installed but cannot set control to 'Automatic' mode."
128        )
129    warnings.warn(
130        "E-beam shutter for FIB milling operations is in auto-mode, which may not insert at certain tilt angles and stage positions. Manual control not available."
131    )
132    return

135def prepare_milling(
136    microscope: tbt.Microscope,
137    application: str,
138    patterning_device: tbt.Device = tbt.Device.ION_BEAM,
139) -> bool:
140    # TODO validation and error checking from TFS
141    # TODO support e-beam patterning via the mill_beam settings
142    """
143    Clear old patterns, assign patterning to ion beam by default, and load the application.
144
145    This function clears old patterns, assigns the patterning device to the specified beam (ion beam by default), and loads the specified application. It validates the patterning device and application file.
146
147    Parameters
148    ----------
149    microscope : tbt.Microscope
150        The microscope object for which to prepare milling.
151    application : str
152        The name of the application file to load.
153    patterning_device : tbt.Device, optional
154        The device to use for patterning (default is tbt.Device.ION_BEAM).
155
156    Returns
157    -------
158    bool
159        True if the preparation is successful.
160
161    Raises
162    ------
163    ValueError
164        If the patterning device is invalid or if the application file is not found on the system.
165    """
166    valid_devices = [tbt.Device.ELECTRON_BEAM, tbt.Device.ION_BEAM]
167    if patterning_device not in valid_devices:
168        raise ValueError(
169            f"Invalid patterning device of '{patterning_device}' requested, only '{[i for i in valid_devices]}' are valid."
170        )
171    microscope.patterning.clear_patterns()
172    microscope.patterning.set_default_beam_type(beam_index=patterning_device.value)
173    if application not in microscope.patterning.list_all_application_files():
174        raise ValueError(
175            f"Invalid application file on this system, there is no patterning application with name: '{application}'."
176        )
177    else:
178        microscope.patterning.set_default_application_file(application)
179
180    return True

327@create_pattern.register
328def _(
329    geometry: tbt.FIBStreamPattern,
330    microscope: tbt.Microscope,
331    **kwargs: dict,
332) -> tbt.StreamPattern:
333    """
334    Create a stream pattern on the microscope.
335
336    Parameters
337    ----------
338    geometry : tbt.FIBStreamPattern
339        The geometry of the stream pattern to create.
340    microscope : tbt.Microscope
341        The microscope object on which to create the pattern.
342    kwargs : dict
343        Additional keyword arguments.
344
345    Returns
346    -------
347    tbt.StreamPattern
348        The created stream pattern.
349    """
350
351    # run image_processing and check that mask file is created
352    input_image_path = kwargs["kwargs"]["input_image_path"]
353    image_processing(
354        geometry=geometry,
355        input_image_path=input_image_path,
356    )
357
358    # get mask
359    mask_path = geometry.mask_file
360    with pil_img.open(mask_path) as mask_img:
361        width, height = mask_img.size
362        mask = np.asarray(mask_img).astype(int)
363
364    stream_def = tbt.StreamPatternDefinition()
365    stream_def.bit_depth = tbt.StreamDepth.BITS_16  # only supported bit depth now
366    dwell_time = geometry.dwell_us * Conversions.US_TO_S
367
368    scale_factor = cs.Constants.stream_pattern_scale / width
369    point_img = cv2.resize(
370        mask,
371        dsize=(int(width * scale_factor), int(height * scale_factor)),
372        interpolation=cv2.INTER_NEAREST,
373    )
374    idx = np.where(point_img == 1)
375    num_points = (
376        len(idx[0]) + 2
377    )  # top right and bottom left corners each have a point to make sure pattern is centered
378
379    # stream pattern is defined by 4 values for each point
380    # [x, y, dwell_time, flags]
381    # flags = 1 --> blank beam
382    # flags = 0 --> use beam
383    stream_def.points = np.zeros(shape=(num_points, 4), dtype=object)
384    stream_def.points[0] = [
385        1,  # x (top left)
386        cs.Constants.stream_pattern_y_shift,  # y (top left)
387        dwell_time,  # dwell
388        1,  # flag (1 means blank the beam)
389    ]
390    flags = 0
391    for i in range(1, num_points - 1):  # start at first point
392        x = idx[1][i - 1] * 16 + 1
393        y = (
394            idx[0][i - 1] * 16 + cs.Constants.stream_pattern_y_shift
395        )  # + (0.17 * (2**stream_def.bit_depth))
396        stream_def.points[i] = [x, y, dwell_time, flags]
397    stream_def.points[-1] = [
398        2**stream_def.bit_depth,  # x (bottom right)
399        2**stream_def.bit_depth
400        - cs.Constants.stream_pattern_y_shift,  # y (bottom right)
401        dwell_time,  # dwell
402        1,  # flag
403    ]
404
405    stream_def.repeat_count = geometry.repeats
406
407    stream_pattern = microscope.patterning.create_stream(
408        center_x=0.0,
409        center_y=0.0,
410        stream_pattern_definition=stream_def,
411    )
412
413    return stream_pattern

416def image_processing(
417    geometry: tbt.FIBStreamPattern,
418    input_image_path: Path,
419) -> bool:
420    """
421    Perform image processing for FIB stream pattern.
422
423    This function runs an image processing script specified by the `recipe_file` in the `geometry` object, using the input image path and outputting the mask file.
424
425    Parameters
426    ----------
427    geometry : tbt.FIBStreamPattern
428        The geometry of the FIB stream pattern, including the `recipe_file` and `mask_file`.
429    input_image_path : Path
430        The path to the input image.
431
432    Returns
433    -------
434    bool
435        True if the image processing is successful.
436
437    Raises
438    ------
439    ValueError
440        If the subprocess call for the script does not execute correctly or if the mask file is not created.
441    """
442    output = subprocess.run(
443        [
444            "python",
445            (geometry.recipe_file).as_posix(),  # recipe_file
446            input_image_path.as_posix(),  # input path,
447            (geometry.mask_file).as_posix(),  # outputpath,
448        ],
449        capture_output=True,
450    )
451    if output.returncode != 0:
452        raise ValueError(
453            f"Subprocess call for script {geometry.recipe_file} using executable 'python' did not execute correctly."
454        )
455    # check for mask file
456    if not geometry.mask_file.exists():
457        raise ValueError(
458            f"Mask file at location {geometry.mask_file} should have been created by image processing recipe but does not exist. Please check your recipe_file script."
459        )
460    # TODO
461    # save masks
462
463    return True

467def mill_operation(
468    step: tbt.Step,
469    fib_settings: tbt.FIBSettings,
470    general_settings: tbt.GeneralSettings,
471    slice_number: int,
472) -> bool:
473    """
474    Perform a milling operation based on the provided step and settings.
475
476    This function performs a milling operation using the specified step, FIB settings, general settings, and slice number.
477
478    Parameters
479    ----------
480    step : tbt.Step
481        The step object containing the operation settings.
482    fib_settings : tbt.FIBSettings
483        The FIB settings object containing the microscope and pattern settings.
484    general_settings : tbt.GeneralSettings
485        The general settings object.
486    slice_number : int
487        The slice number for the operation.
488
489    Returns
490    -------
491    bool
492        True if the milling operation is successful.
493
494    Raises
495    ------
496    ValueError
497        If the ion image for selected area milling is not found.
498    """
499    microscope = fib_settings.microscope
500
501    shutter_control(microscope=microscope)
502    # prepare beam
503    img.imaging_device(microscope=microscope, beam=fib_settings.mill_beam)
504    # set milling application and device
505
506    prepare_milling(
507        microscope=microscope,
508        application=fib_settings.pattern.application,
509    )
510
511    # get expected path of the fib image
512    if fib_settings.pattern.type != tbt.FIBPatternType.SELECTED_AREA:
513        input_image_path = None
514    else:
515        input_image_path = Path.join(
516            general_settings.exp_dir,
517            step.name,
518            f"{slice_number:04}.tif",
519        )
520        if not input_image_path.exists():
521            raise ValueError(
522                f"Ion image for selected area milling was not found at '{input_image_path}'."
523            )
524
525    # make the pattern
526    pattern = create_pattern(
527        fib_settings.pattern.geometry,
528        microscope=microscope,
529        kwargs={"input_image_path": input_image_path},
530    )
531
532    microscope.patterning.run()
533
534    return True

def mill_operation( step: pytribeam.types.Step, fib_settings: pytribeam.types.FIBSettings, general_settings: pytribeam.types.GeneralSettings, slice_number: int) -> bool: 
468def mill_operation(
469    step: tbt.Step,
470    fib_settings: tbt.FIBSettings,
471    general_settings: tbt.GeneralSettings,
472    slice_number: int,
473) -> bool:
474    """
475    Perform a milling operation based on the provided step and settings.
476
477    This function performs a milling operation using the specified step, FIB settings, general settings, and slice number.
478
479    Parameters
480    ----------
481    step : tbt.Step
482        The step object containing the operation settings.
483    fib_settings : tbt.FIBSettings
484        The FIB settings object containing the microscope and pattern settings.
485    general_settings : tbt.GeneralSettings
486        The general settings object.
487    slice_number : int
488        The slice number for the operation.
489
490    Returns
491    -------
492    bool
493        True if the milling operation is successful.
494
495    Raises
496    ------
497    ValueError
498        If the ion image for selected area milling is not found.
499    """
500    microscope = fib_settings.microscope
501
502    shutter_control(microscope=microscope)
503    # prepare beam
504    img.imaging_device(microscope=microscope, beam=fib_settings.mill_beam)
505    # set milling application and device
506
507    prepare_milling(
508        microscope=microscope,
509        application=fib_settings.pattern.application,
510    )
511
512    # get expected path of the fib image
513    if fib_settings.pattern.type != tbt.FIBPatternType.SELECTED_AREA:
514        input_image_path = None
515    else:
516        input_image_path = Path.join(
517            general_settings.exp_dir,
518            step.name,
519            f"{slice_number:04}.tif",
520        )
521        if not input_image_path.exists():
522            raise ValueError(
523                f"Ion image for selected area milling was not found at '{input_image_path}'."
524            )
525
526    # make the pattern
527    pattern = create_pattern(
528        fib_settings.pattern.geometry,
529        microscope=microscope,
530        kwargs={"input_image_path": input_image_path},
531    )
532
533    microscope.patterning.run()
534
535    return True
